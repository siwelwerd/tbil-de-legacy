\moduleQuestion{How can we use numerical approximation methods to apply and solve unsolvable ODEs?}
\begin{moduleStandards}
	\standard{First Order Existence and Uniqueness}{determine when a unique solution exists for a  first order ODE}
	\standard{Second Order Linear Existence and Uniqueness}{determine when a unique solution exists for a second order linear ODE}
		\standard{Systems Existence and Uniqueness}{determine when a unique solution exists for a system of first order ODEs}
		  \standard{Euler's method for first order ODES}{use Euler's method to find approximate solution to first order ODEs}
		\standard{Euler's method for systems}{use Euler's method to find approximate solutions to systems of first order ODEs}
\end{moduleStandards}
